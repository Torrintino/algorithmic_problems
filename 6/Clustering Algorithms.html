def csr_to_networkit(A):
        """ Converts an adjacency matrix to a NetworKit graph. """
        g=nk.graph.Graph(A.shape[0])
        row_idx, col_idx = np.nonzero(sparse.triu(A))
        for u, v in zip(row_idx, col_idx):
            g.addEdge(u, v)

        return g
